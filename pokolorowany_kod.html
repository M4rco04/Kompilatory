def reverse_number(n):
    reverse_n = 0
    i=10
    while n>0:
        reverse_n = reverse_n*i + n%10
        n//=10
    return reverse_n

def pitagoras_triangles(n):
    triangles = []
    for a in range(1, n):
        for b in range(a, n):
            c = a*a + b*b
            if c**(1/2) == int(c**(1/2)) and c**(1/2)<n:
                c = int(c**(1/2))
                triangles.append((a, b, c))
            else:
                pass
    return triangles

def reverse_polish_notation(list_phrase):
    help = []
    for i in list_phrase:
        if i not in ['+','-','*','/']:
            help.append(i)
        else:
            a = help[len(help)-2]
            b = help[len(help)-1]
            help = help[:len(help)-2]
            if i == '+':
                help.append(a+b)
            elif i == '-':
                help.append(a-b)
            elif i=='/':
                help.append(a/b)
            else:
                help.append(a*b)
    return help[0]

def factorial(n):
    if n==0 or n==1:
        return 1
    return n*factorial(n-1)

def rec_searching_maksimum(list, a=0, maks=0):
    if a==0:
        return rec_searching_maksimum(list, a+1, list[a])
    if a < len(list)-1:
        if list[a] > maks:
            return rec_searching_maksimum(list, a+1, list[a])
        return rec_searching_maksimum(list, a+1, maks)
    if maks > list[a]:
        return maks
    return list[a]

def rec_list_searching_for_value(list, a, b, value):
    pivot = (a+b)//2 #Przeszukiwany indeks
    if a <= b:
        if list[pivot] < value:
            a = pivot + 1
            return rec_list_searching_for_value(list, a, b, value)
        if list[pivot] > value:
            b = pivot - 1
            return rec_list_searching_for_value(list, a, b, value)
        else:
            return pivot
    else:
        return None
